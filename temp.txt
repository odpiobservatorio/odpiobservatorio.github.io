        var tooltip = L.tooltip({ permanent: true, interactive: true})
        .setLatLng([LatB, LngB])
        .setContent(LbEdit)
        .addTo(map);
        map.on('mouseup',
        function (event) {
            var a = event.latlng.lat
            var b = event.latlng.lng
            tooltip.setLatLng(new L.LatLng(a, b));
            
        });
    var tooltipElement = tooltip.getElement();
    var draggable = new L.Draggable(tooltipElement);
    draggable.enable();
